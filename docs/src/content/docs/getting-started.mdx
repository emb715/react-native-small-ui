## Installation

React Native Small UI uses a modular architecture. Install only what you need to keep your bundle size minimal.

### Core Package (Required)

The core package provides `createComponent` and essential utilities with minimal footprint (~15KB).

## Quick Start

### Minimal Setup (Core Only)

Get started by adding the `useSmallUI` hook in your App and create styled components.

```js
import * as React from 'react';
import { View, Text, TouchableOpacity } from 'react-native';
import { useSmallUI, createComponent } from 'react-native-small-ui';

// Create styled components
const Box = createComponent(View, {
  padding: 16,
  borderRadius: 8,
  _light: {
    backgroundColor: '#f5f5f5',
  },
  _dark: {
    backgroundColor: '#1a1a1a',
  },
  _ios: {
    shadowOpacity: 0.1,
  },
  _android: {
    elevation: 2,
  },
});

const Button = createComponent(TouchableOpacity, {
  padding: 12,
  borderRadius: 6,
  backgroundColor: '#007AFF',
  alignItems: 'center',
});

function App() {
  return (
    <View>
      <Box marginTop={112}>
        <Text>Hello Small UI!</Text>
        <Button>
          <Text style={{ color: '#fff' }}>Click me</Text>
        </Button>
      </Box>
    </View>
  );
}

export default function Main() {
  useSmallUI();
  return <App />;
}
```

### With Color Mode

Add programmatic theme control:

```js
import { useSmallUI, createComponent } from 'react-native-small-ui';
import { useColorMode, toggleColorScheme } from 'react-native-small-ui/colormode';
import { View, Text, TouchableOpacity } from 'react-native';

const Box = createComponent(View, {
  padding: 16,
  _light: { backgroundColor: '#fff' },
  _dark: { backgroundColor: '#000' },
});

function ThemeToggle() {
  const { colorMode } = useColorMode();

  return (
    <Box>
      <Text>Current mode: {colorMode}</Text>
      <TouchableOpacity onPress={toggleColorScheme}>
        <Text>Toggle Theme</Text>
      </TouchableOpacity>
    </Box>
  );
}

export default function Main() {
  useSmallUI();
  return <ThemeToggle />;
}
```

### With Responsive Utilities

Build responsive layouts:

```js
import { useSmallUI, createComponent } from 'react-native-small-ui';
import { useBreakPointValue } from 'react-native-small-ui/utils';
import { View, Text } from 'react-native';

// Create component outside render
const Box = createComponent(View, {
  backgroundColor: '#f5f5f5',
});

function ResponsiveBox() {
  const padding = useBreakPointValue({
    default: 8,
    sm: 12,
    md: 16,
    lg: 24,
  });

  return (
    <Box padding={padding}>
      <Text>Responsive padding based on screen size</Text>
    </Box>
  );
}

export default function Main() {
  useSmallUI();
  return <ResponsiveBox />;
}
```

### With Theme System

Use semantic colors and design tokens:

```js
import { useSmallUI, createComponent } from 'react-native-small-ui';
import { useTheme } from 'react-native-small-ui/theme';
import { View, Text, TouchableOpacity } from 'react-native';

// Create component outside render
const Button = createComponent(TouchableOpacity, {
  borderRadius: 8,
});

function ThemedButton() {
  const theme = useTheme();

  return (
    <Button
      padding={theme.space?.[4]}
      _light={{ backgroundColor: theme.colors.light.primary }}
      _dark={{ backgroundColor: theme.colors.dark.primary }}
    >
      <Text style={{ color: '#fff' }}>Themed Button</Text>
    </Button>
  );
}

export default function Main() {
  useSmallUI();
  return <ThemedButton />;
}
```
